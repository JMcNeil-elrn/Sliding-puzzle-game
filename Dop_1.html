  <script>

    document.addEventListener('DOMContentLoaded', () => {

      const fallZone = document.querySelector('.fall-zone');

      const cooler = document.querySelector('.cooler');

      const scoreDisplay = document.querySelector('.score');

      const qrCodeContainer = document.querySelector('.qr-code-container');

      const startButton = document.getElementById('startGame');

      const showQRCodeButton = document.getElementById('showQRCode');

      const icons = { beer: '🍺', snake: '🐍' };

      let score = 0;

      let gameInterval;


      // Helper: Create falling items (beer or snake)

      function createItem() {

        const item = document.createElement('div');

        item.classList.add('item');

        

        // Randomize item type (beer or snake)

        const isBeer = Math.random() < 0.7; // 70% chance for beer

        item.textContent = isBeer ? icons.beer : icons.snake;


        // Random position along the top width

        item.style.left = Math.random() * 90 + 'vw';

        item.style.top = '-50px';


        // Falling animation

        const fallDuration = Math.random() * 3 + 2; // Duration between 2-5 seconds

        const startTime = Date.now();

        const fallInterval = setInterval(() => {

          const elapsed = Date.now() - startTime;

          const progress = elapsed / (fallDuration * 1000);


          // Move item down

          item.style.transform = `translateY(${progress * 100}vh)`;


          // Remove item when it exits screen

          if (progress >= 1) {

            clearInterval(fallInterval);

            if (fallZone.contains(item)) fallZone.removeChild(item);

          }

        }, 30);


        // Handle click on items

        item.addEventListener('click', () => {

          clearInterval(fallInterval);

          handleItemClick(item);

          if (fallZone.contains(item)) fallZone.removeChild(item);

        });


        fallZone.appendChild(item);

      }


      // Function to handle a clicked item

      function handleItemClick(item) {

        const itemType = item.textContent;


        if (itemType === icons.beer) {

          score++;

          updateScore();

          updateCooler(itemType);

        }

      }


      // Update score display

      function updateScore() {

        scoreDisplay.textContent = `Beers Collected: ${score}`;

        if (score >= 35) {

          endGame();

        }

      }


      // Add beer to cooler

      function updateCooler(emoji) {

        const beerIcon = document.createElement('span');

        beerIcon.textContent = emoji;

        cooler.appendChild(beerIcon);

      }


      // End the game

      function endGame() {

        clearInterval(gameInterval);

        startButton.disabled = true;

        showQRCodeButton.style.display = 'inline-block';

      }


      // Start the game

      startButton.addEventListener('click', () => {

        resetGame();

        gameInterval = setInterval(createItem, 800); // Spawn items every 800ms

      });


      // Reset game function

      function resetGame() {

        score = 0;

        cooler.innerHTML = '';

        updateScore();

        qrCodeContainer.style.display = 'none';

        startButton.disabled = false;

      }


      // Show QR Code button logic

      showQRCodeButton.addEventListener('click', () => {

        qrCodeContainer.style.display = 'block';

      });

    });

  </script>
